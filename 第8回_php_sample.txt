<?php

class Myclass {
//ここに関数とか定数を書く
 public $aaa = "123";//外から見れる
}

//使い方
$aaa = new Myclass(); //$aaa = new Myclass;でもOK
//newしない場合の書き方
echo MyClass::$aaa;//上のpublicのあとにstatic(静的)って入れないとだめ　sataticって書いてなかったらnewして呼ぶ

class Myclass {
 private $aaa = "123";//外からアクセスできない

 public static function efg (){
   return "456";
 }
 echo MyClass::efg();//456が出力される

}

class Myclass {
 private static $aaa = "123";//外からアクセスできない

 public static function efg (){
   return self::$aaa;
 }
 echo MyClass::efg();//123が出力される
echo MyClass::$aaa//error
}

<!--下の2つが対-->

class Myclass{ 

 protected static $aaa = "123";
 public $bbb = "abc";

 public function hij(){
     return "000";
 }

 public static function efg (){
   return self::$aaa;//staticって書く場合selfとか::の書き方を使う
 }
$my = new MyClass;
echo $my -> hij(); 
}

class Myclass{ 

 protected static $aaa = "123";
 public $bbb = "abc";

 public function hij(){
     return $this -> bbb;//staticって書かない場合、$thisを使う
 }

 public static function efg (){
   return self::$aaa;
 }
$my = new MyClass;
echo $my -> hij(); 
}

<!--区切り-->

class Myclass{ 
    public static $aaa = "123";

    public function hij(){
        return $this->bbb;
    }
    public static function efg(){
        return self::$aaa;
    }
 }

//newのやつ
$my = new MyClass;

//動く
echo $my->bbb;
echo $my->hij();

//動かない
echo $my->aaa;

//動かんと思ったけど動いた
echo $my->efg();

//staticのやつ
//動く
echo MyClass::$aaa;
echo MyClass::efg();

//動かない
echo MyClass::$bbb;
echo MyClass::hij();
